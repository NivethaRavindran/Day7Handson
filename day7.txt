package com.day7;

import java.util.Scanner;
class Runs{
	void displayRuns(int x){
		int sum=0,avg=0,k;
		Scanner sn=new Scanner(System.in);
		System.out.println("The runs are:");
		for(int j=0;j<x;j++){
			k=sn.nextInt();
			sum=sum+k;
			
		}
		System.out.println("The total runs:"+sum);
		
		avg=sum/x;
		System.out.println("Avg run is:"+avg);
		
	}
}

package com.day7;

public class Program2 {
	public static void main(String[] args){

		Scanner sn=new Scanner(System.in);
		System.out.println("Enter the total matches by raina:");
		runs=sn.nextInt();
		System.out.println("Enter runs:");
		Set<Integer> i=new TreeSet<>();
		for(int j=0;j<runs;j++){
			int k=sn.nextInt();
			i.add(k);
		}
		System.out.println("The sorted runs");
		for(int x:i){

	}
